import sys

def Bubblesort(n, a):
    numberofswaps = 0
    for j in range(n):
        x = (n-1)-j
        for i in range(x):
            # print(i, a[i], a[i+1])
            if(a[i]>a[i+1]):
                # temp = a[i]
                # a[i] = a[i+1]
                # a[i+1] = temp
                a[i], a[i+1] = a[i+1], a[i]
                numberofswaps += 1
    return(a, numberofswaps)        


n = int(input().strip())
a = list(map(int, input().strip().split(' ')))
sortedArray, swaps = Bubblesort(n, a)
print("Array is sorted in",swaps)
print("First Element:",sortedArray[0])
print("Last Element:",sortedArray[-1])
